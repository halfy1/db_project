        // Переключение вкладок проекта
        function showProjectTab(tabId) {
            document.querySelectorAll('.project-tab-content').forEach(tab => {
                tab.classList.add('hidden');
            });
            document.getElementById(tabId).classList.remove('hidden');
            
            // Обновить активную вкладку
            document.querySelectorAll('.project-tab').forEach(tab => {
                tab.classList.remove('border-crm-primary', 'text-crm-primary', 'dark:text-crm-secondary');
                tab.classList.add('border-transparent', 'text-gray-500', 'hover:text-gray-700', 'hover:border-gray-300', 'dark:text-gray-400', 'dark:hover:text-gray-300');
            });
            
            const activeTab = document.querySelector(`.project-tab[onclick="showProjectTab('${tabId}')"]`);
            activeTab.classList.add('border-crm-primary', 'text-crm-primary', 'dark:text-crm-secondary');
            activeTab.classList.remove('border-transparent', 'text-gray-500', 'hover:text-gray-700', 'hover:border-gray-300', 'dark:text-gray-400', 'dark:hover:text-gray-300');
            
            // Инициализировать графики если открыта вкладка аналитики
            if (tabId === 'analytics' && !window.teamActivityChart) {
                initProjectCharts();
            }
        }
        
        // Переключение темы
        function toggleTheme() {
            const html = document.documentElement;
            html.classList.toggle('dark');
            localStorage.setItem('darkMode', html.classList.contains('dark'));
            
            // Обновить иконку
            const themeIcon = document.getElementById('themeToggle');
            if (html.classList.contains('dark')) {
                themeIcon.innerHTML = '<i class="fas fa-sun"></i>';
            } else {
                themeIcon.innerHTML = '<i class="fas fa-moon"></i>';
            }
            
            // Перерисовать графики если они есть
            if (window.teamActivityChart) {
                window.teamActivityChart.destroy();
                window.bugsChart.destroy();
                window.tasksChart.destroy();
                initProjectCharts();
            }
        }
        
        // Проверка темы при загрузке
        function checkTheme() {
            const darkMode = localStorage.getItem('darkMode') === 'true' || 
                            (!localStorage.getItem('darkMode') && window.matchMedia('(prefers-color-scheme: dark)').matches);
            
            if (darkMode) {
                document.documentElement.classList.add('dark');
                document.getElementById('themeToggle').innerHTML = '<i class="fas fa-sun"></i>';
            }
        }
        
        // Инициализация при загрузке
        document.addEventListener('DOMContentLoaded', function() {
            checkTheme();
            document.getElementById('themeToggle').addEventListener('click', toggleTheme);
            
            // Инициализация графиков если открыта вкладка аналитики
            if (document.getElementById('analytics').classList.contains('hidden') === false) {
                initProjectCharts();
            }
        });
